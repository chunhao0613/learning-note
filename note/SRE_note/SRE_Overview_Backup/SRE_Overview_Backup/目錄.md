```dataviewjs
// 1. 設定你的資料夾路徑
const folderPath = "note/SRE_note/SRE_Overview_Backup/SRE_Overview_Backup";

// 2. 取得 Vault 中所有的檔案
const files = app.vault.getFiles()
    // 3. 過濾出符合路徑且副檔名為 md 的檔案
    .filter(file => file.path.startsWith(folderPath) && file.extension === "md")
    // 🌟 4. 排序：依照檔案名稱 (file.name) 進行升冪排序
    .sort((a, b) => a.name.localeCompare(b.name, undefined, {numeric: true, sensitivity: 'base'}));

// 5. 將結果轉換為 Obsidian 連結並以清單顯示
dv.list(files.map(file => dv.fileLink(file.path)));
```
